class Vec2
{
	public attribute x : integer;
	public attribute y : integer;
};

class Vec3
{
	public attribute v : Vec2;
	public attribute z : integer;
};

class A
{
	public attribute vec : Vec3;
};

function main() => void
{
	localvar a : A;
	a.vec.v.x = 1;
	a.vec.v.y = 3;
	a.vec.z = 2;
	write(a.vec.v.x + a.vec.v.y + a.vec.z);
}
