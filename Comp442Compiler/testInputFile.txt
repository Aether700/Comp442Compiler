class Vec2
{
    public attribute x : float;
    public attribute y : float;

    constructor : (x : float, y : float);
};

function Vec2::constructor(x : float, y : float)
{
    self.x = x;
    self.y = y;
}

class Vec3 isa Vec2
{
    public attribute z : float;
    constructor : (x : float, y : float, z : float);
};

function Vec3::constructor(x : float, y : float, z : float)
{
    self.x = x;
    self.y = y;
    self.z = z;
}
