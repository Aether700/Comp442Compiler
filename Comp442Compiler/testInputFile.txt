class A
{
	
	public attribute x : integer;
	public attribute y : integer;

	public constructor:(x_ : integer, y_ : integer);
	public constructor:(n : integer);
};

function A::constructor(n : integer)
{
	self.x = n;
	self.y = n;
}

function A::constructor(x_ : integer, y_ : integer)
{
	self.x = x_;
	self.y = y_;
}

function main()=>void
{
	localvar a : A(15);
	localvar b : A(5, 3);
	
	// should write 153
	write(a.y);
	write(b.y);
}
