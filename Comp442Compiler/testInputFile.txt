class B
{
	public attribute z : integer;
};

class C
{
	public attribute w : integer;
};

class A isa B, C
{
	public attribute x : integer;
	public attribute y : integer;
};

function main() => void
{
	localvar a : A;
	a.x = 4;
	a.y = 5;
	a.z = 6;
	a.w = 7;
	write(a.z);
	write(a.w);
	write(a.z - a.y);
}
